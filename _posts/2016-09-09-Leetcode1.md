# Leetcode No.1 Two Sum <br/>
注意vector传入参数的方式:

	#include<iostream>
	#include<vector>
	using namespace std;

	class Solution {
	public:
   	  vector<int> twoSum(vector<int>& nums, int target) {
        //cout<<nums.size()<<endl;
        vector<int> result;
        int size = nums.size();
        for (int i = 0 ; i < size ; i++) {
            for (int j = i+1 ; j < size ; j++) {
                if (nums[i]+nums[j] == target) {
                    result.push_back(i);
                    result.push_back(j);
                    cout<<"nums["<<i<<"]:"<<nums[i]<<endl;
                    cout<<"nums["<<j<<"]:"<<nums[j]<<endl;
                    break;
                }
            }
        }
    	return result;
    	}
	};
	int main() {
    	Solution S;
    	vector<int> nums;
    	nums.push_back(2);
    	nums.push_back(7);
    	nums.push_back(9);
    	nums.push_back(11);
    	int num = 18;
    	S.twoSum(nums, num);
    	return 0;
 	}
